jsasync function executeDerivTrade(contractType, stakeAmount, accountId, accessToken) {
  // Step 1: Get fresh OTP WebSocket URL
  let wsUrl;
  try {
    const otpResponse = await fetch('/api/deriv/ws-token', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ accountId, accessToken })
    });

    if (!otpResponse.ok) {
      const err = await otpResponse.json();
      showTradeError('Failed to get trade session: ' + (err.error?.message || 'Unknown error'));
      return;
    }

    const { wsUrl: url } = await otpResponse.json();
    wsUrl = url;
  } catch (e) {
    showTradeError('Network error getting trade session.');
    return;
  }

  // Step 2: Open the authenticated WebSocket — NO authorize message needed
  const ws = new WebSocket(wsUrl);
  let proposalId = null;

  ws.onopen = () => {
    // Step 3: Send proposal request immediately on open
    ws.send(JSON.stringify({
      proposal: 1,
      amount: stakeAmount,
      basis: 'stake',
      contract_type: contractType,   // "CALL" or "PUT"
      currency: 'USD',
      duration: 2,
      duration_unit: 'm',
      symbol: 'R_75',
      req_id: 1
    }));
  };

  ws.onmessage = (event) => {
    const msg = JSON.parse(event.data);

    // Handle errors from Deriv
    if (msg.error) {
      showTradeError('Deriv error: ' + msg.error.message);
      ws.close();
      return;
    }

    // Step 4: Proposal received — extract ID and buy immediately
    if (msg.msg_type === 'proposal') {
      proposalId = msg.proposal.id;
      ws.send(JSON.stringify({
        buy: proposalId,
        price: stakeAmount,
        req_id: 2
      }));
    }

    // Step 5: Buy confirmed — record result and close
    if (msg.msg_type === 'buy') {
      const result = {
        contractId: msg.buy.contract_id,
        executionPrice: msg.buy.buy_price,
        longcode: msg.buy.longcode,
        entrySpot: msg.buy.start_time
      };

      // Log result to your signal history / DB
      recordTradeExecution(result);

      // Show confirmation in UI
      showTradeSuccess(`Trade placed: ${contractType} @ $${result.executionPrice}`);

      // Close this WebSocket — it was single-use for this trade
      ws.close();
    }
  };

  ws.onerror = (err) => {
    showTradeError('WebSocket connection to Deriv failed. Retrying...');
    ws.close();
  };

  ws.onclose = () => {
    // Connection closed normally — no action needed
  };
}
